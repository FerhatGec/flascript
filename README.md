     * Fast, functional scripting language.

## A taste of FlaScript
```go
import(<stdlib/output/Colorized.flsh>) -> color <-

defin[slashn] -> print(newline) <-

func -> PrintWithColor() {
    put[color -> RED <-]
    print(string) -> "Whoa! Hello developer!" 
    put[color -> BLACK_COLOR <-]
    put[slashn]
} PrintWithColor;

main() -> main {
    func -> PrintWithColor()
} main;
```
